Column 정보

  • Regional.Indicator : 지역 표시

  • Ladder.score : 행복지수

  • Standard.error.of.ladder.score : 행복지수의 표준 오차

  • upperwhisker : 최댓값

  • lowerwhisker : 최솟값

  • Logged.GDP.per.capita : 인당 GDP(log scale)

  • Social.support : 사회적 지원

  • Healthy.life.expectancy : 기대 수명

  • Freedom.to.make.life.choices : 삶의 선택에 대한 자유

  • Generosity : 관대함

  • Perceptions.of.corruption : 부패에 대한 인식

  • Ladder.score.in.Dystopia : 디스토피아에서의 행복지수

  • Explained.by..Log.GDP.per.capita : 인당 GDP에 의해 설명

  • Explained.by..Social.support : 사회적 지지에 의해 설명

  • Explained.by..Healthy.life.expectanc : 기대 수명에 의해 설명

  • Explained.by..Freedom.to.make.life.choices : 삶의 선택에 대한 자유에 의해 설명

  • Explained.by..Generosity : 관대함에 의해 설명

  • Explained.by..Perceptions.of.corruption : 부패에 대한 인식에 의해 설명

  • Dystopia…residual : 디스토피아에서의 행복지수 잔차

dimensions <- c("Ladder.score", "Logged.GDP.per.capita", "Social.support", "Healthy.life.expectancy", "Freedom.to.make.life.choices", "Generosity", "Perceptions.of.corruption")

  unique()

df_long <- df %>% 
  select(country = Country.name, all_of(dimensions)) %>%
  mutate(absence_of_corruption = 1 - Perceptions.of.corruption) %>%
  pivot_longer(cols = c(all_of(dimensions), "absence_of_corruption"), names_to = "dimension", values_to = "score") %>%
  filter(dimension != "Perceptions.of.corruption")

df_tranformed <- df_long %>%
    group_by(dimension) %>%
    mutate(min_value = min(score),

    ungroup()

donut_plot <- function(df, title = '', subtitle = '', caption = '') {
ggplot(df) +
    geom_rect(aes(ymax = 1, ymin = 0, xmax = 2, xmin=1.2, fill = "base"))  +
    geom_rect(aes(ymax = pct, ymin = 0, xmax = 2.2, xmin = 1.2, fill = 'main')) +
    geom_text(x = 0, y = 0, label = paste0(round(df$pct*100,0),'%'), size = 12) + 
    coord_polar(theta = 'y') +
    xlim(c(0,2.2)) + 
    scale_fill_manual(values = c("#e3e2e1", "#7FB185")) +
    labs(title = title, subtitle = subtitle, caption = caption) +
    theme_void() + 
    theme(plot.title = element_text(size=15, face = 'bold'),
          plot.subtitle = element_text(size = 12, hjust = 0.5),
          plot.caption = element_text(size = 9),
